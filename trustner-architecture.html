üóÑÔ∏è Database Schema Design

users

id UUID PRIMARY KEY
email VARCHAR(255) UNIQUE
phone VARCHAR(15) UNIQUE
first_name VARCHAR(100)
last_name VARCHAR(100)
role ENUM('customer', 'posp', 'advisor', 'admin')
is_active BOOLEAN DEFAULT TRUE
created_at TIMESTAMP

posp_profiles

id UUID PRIMARY KEY
user_id UUID FOREIGN KEY -> users(id)
posp_code VARCHAR(50) UNIQUE
certification_status ENUM('pending', 'certified', 'expired')
license_number VARCHAR(100)
license_expiry DATE
hierarchy_level ENUM('SP', 'BQP', 'RM')
parent_id UUID FOREIGN KEY -> posp_profiles(id)

products

id UUID PRIMARY KEY
product_code VARCHAR(50) UNIQUE
name VARCHAR(255)
category ENUM('motor', 'health', 'life')
insurer_id UUID FOREIGN KEY -> insurers(id)
features JSONB
is_active BOOLEAN DEFAULT TRUE

quotes

id UUID PRIMARY KEY
quote_number VARCHAR(50) UNIQUE
user_id UUID FOREIGN KEY -> users(id)
product_id UUID FOREIGN KEY -> products(id)
premium_amount DECIMAL(10,2)
sum_insured DECIMAL(12,2)
valid_until TIMESTAMP
status ENUM('active', 'expired', 'converted')

leads

id UUID PRIMARY KEY
lead_number VARCHAR(50) UNIQUE
user_id UUID FOREIGN KEY -> users(id)
quote_id UUID FOREIGN KEY -> quotes(id)
assigned_to UUID FOREIGN KEY -> users(id)
status ENUM('new', 'in_progress', 'converted', 'closed')
source VARCHAR(100)
context_data JSONB

policies (Renewal Readiness)

id UUID PRIMARY KEY
policy_number VARCHAR(100) UNIQUE
user_id UUID FOREIGN KEY -> users(id)
product_id UUID FOREIGN KEY -> products(id)
start_date DATE
expiry_date DATE
grace_period_days INTEGER
renewal_eligible BOOLEAN

audit_logs

id UUID PRIMARY KEY
user_id UUID FOREIGN KEY -> users(id)
action VARCHAR(100)
entity_type VARCHAR(50)
entity_id UUID
old_values JSONB
new_values JSONB
timestamp TIMESTAMP (immutable)
